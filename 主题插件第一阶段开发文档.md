将当前硬编码的主题系统重构为**插件化架构**，实现：
- ✅ 主题配置从硬编码转为数据驱动
- ✅ 支持主题注册、切换、预览
- ✅ 保留现有的樱花特效
- ✅ 保持 UI 完全不变

#### 2.2.2 注册表类

```typescript
// src/registry/ThemeRegistry.ts

import type { ThemeManifest, ThemeColors, MoodColor } from '../types/theme';

/**
 * 主题注册表
 * 
 * 职责：
 * 1. 管理所有已注册的主题
 * 2. 提供主题查询接口
 * 3. 通知订阅者主题变化
 */
class ThemeRegistry {
  private themes: Map<string, ThemeManifest> = new Map();
  private listeners: Set<() => void> = new Set();
  
  /**
   * 注册主题
   */
  register(theme: ThemeManifest): void {
    if (this.themes.has(theme.id)) {
      console.warn(`[ThemeRegistry] 主题 ${theme.id} 已存在，将被覆盖`);
    }
    this.themes.set(theme.id, theme);
    this.notify();
  }
  
  /**
   * 批量注册
   */
  registerBatch(themes: ThemeManifest[]): void {
    themes.forEach(theme => this.register(theme));
  }
  
  /**
   * 注销主题
   */
  unregister(themeId: string): boolean {
    const result = this.themes.delete(themeId);
    if (result) this.notify();
    return result;
  }
  
  /**
   * 获取主题
   */
  get(themeId: string): ThemeManifest | undefined {
    return this.themes.get(themeId);
  }
  
  /**
   * 获取所有主题
   */
  getAll(): ThemeManifest[] {
    return Array.from(this.themes.values());
  }
  
  /**
   * 检查主题是否存在
   */
  has(themeId: string): boolean {
    return this.themes.has(themeId);
  }
  
  /**
   * 订阅变化
   */
  subscribe(listener: () => void): () => void {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  }
  
  /**
   * 通知所有订阅者
   */
  private notify(): void {
    this.listeners.forEach(listener => listener());
  }
}

// 单例导出
export const themeRegistry = new ThemeRegistry();
```

```typescript
/**
 * src/stores/useThemeStore.ts
 * 
 * 主题状态管理
 * 从硬编码重构为使用注册表
 */

import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { themeRegistry } from '../registry/ThemeRegistry';
import { initBuiltInThemes } from '../registry/builtInThemes';
import type { ThemeManifest } from '../types/theme';

// 初始化内置主题
initBuiltInThemes();

interface ThemeState {
  // 当前主题ID
  themeId: string;
  
  // 预览主题ID（临时）
  previewThemeId: string | null;
  
  // 设置主题
  setTheme: (themeId: string) => void;
  
  // 预览主题
  previewTheme: (themeId: string | null) => void;
  
  // 获取当前主题
  getCurrentTheme: () => ThemeManifest | undefined;
  
  // 获取预览主题
  getPreviewTheme: () => ThemeManifest | undefined;
  
  // 获取所有主题
  getAllThemes: () => ThemeManifest[];
}

export const useThemeStore = create<ThemeState>()(
  persist(
    (set, get) => ({
      themeId: 'moodify-theme-terracotta',
      previewThemeId: null,
      
      setTheme: (themeId: string) => {
        if (!themeRegistry.has(themeId)) {
          console.warn(`[ThemeStore] 主题 ${themeId} 不存在`);
          return;
        }
        set({ themeId, previewThemeId: null });
      },
      
      previewTheme: (themeId: string | null) => {
        if (themeId && !themeRegistry.has(themeId)) {
          console.warn(`[ThemeStore] 预览主题 ${themeId} 不存在`);
          return;
        }
        set({ previewThemeId: themeId });
      },
      
      getCurrentTheme: () => {
        const state = get();
        const id = state.previewThemeId || state.themeId;
        return themeRegistry.get(id);
      },
      
      getPreviewTheme: () => {
        const state = get();
        if (!state.previewThemeId) return undefined;
        return themeRegistry.get(state.previewThemeId);
      },
      
      getAllThemes: () => {
        return themeRegistry.getAll();
      },
    }),
    {
      name: 'moodify-theme-v2-storage',
      partialize: (state) => ({ themeId: state.themeId }),
    }
  )
);

// 便捷Hook
export const useCurrentTheme = () => {
  return useThemeStore((state) => state.getCurrentTheme());
};

export const useAllThemes = () => {
  return useThemeStore((state) => state.getAllThemes());
};
```

```typescript
/**
 * src/engine/ThemeEngine.ts
 * 
 * 主题引擎 - 负责将主题配置应用到 DOM
 */

import type { ThemeManifest } from '../types/theme';

class ThemeEngine {
  private currentTheme: ThemeManifest | null = null;
  
  /**
   * 应用主题
   */
  apply(theme: ThemeManifest): void {
    this.currentTheme = theme;
    
    const root = document.documentElement;
    const colors = theme.colors;
    
    // 设置 CSS 变量
    root.style.setProperty('--theme-bg-base', colors.bgBase);
    root.style.setProperty('--theme-bg-surface', colors.bgSurface);
    root.style.setProperty('--theme-bg-elevated', colors.bgElevated);
    
    root.style.setProperty('--theme-text-primary', colors.textPrimary);
    root.style.setProperty('--theme-text-secondary', colors.textSecondary);
    root.style.setProperty('--theme-text-muted', colors.textMuted);
    
    root.style.setProperty('--theme-accent', colors.accent);
    root.style.setProperty('--theme-border', colors.border);
    root.style.setProperty('--theme-border-hover', colors.borderHover);
    
    root.style.setProperty('--theme-coil-color', colors.coil.color);
    root.style.setProperty('--theme-coil-gradient', colors.coil.gradient);
    
    root.style.setProperty('--theme-lost-color', colors.lost.color);
    root.style.setProperty('--theme-lost-gradient', colors.lost.gradient);
    
    root.style.setProperty('--theme-awaken-color', colors.awaken.color);
    root.style.setProperty('--theme-awaken-gradient', colors.awaken.gradient);
    
    root.style.setProperty('--theme-expand-color', colors.expand.color);
    root.style.setProperty('--theme-expand-gradient', colors.expand.gradient);
    
    // 自定义 CSS 变量
    if (theme.cssVariables) {
      Object.entries(theme.cssVariables).forEach(([key, value]) => {
        root.style.setProperty(key, value);
      });
    }
  }
  
  /**
   * 获取当前主题
   */
  getCurrentTheme(): ThemeManifest | null {
    return this.currentTheme;
  }
  
  /**
   * 重置为默认主题
   */
  reset(): void {
    this.currentTheme = null;
  }
}

export const themeEngine = new ThemeEngine();
```

```typescript
/**
 * src/components/shared/ThemeSwitcher.tsx
 * 
 * 主题切换器
 * 使用注册表获取主题列表
 */

import { useState, useRef, useEffect } from 'react';
import { useThemeStore } from '../../stores/useThemeStore';
import { cn } from '../../lib/utils';
import type { ThemeManifest } from '../../types/theme';

export default function ThemeSwitcher() {
  const { themeId, setTheme, getAllThemes } = useThemeStore();
  const themes = getAllThemes();
  
  const [position, setPosition] = useState({ x: window.innerWidth - 80, y: 20 });
  const [isDragging, setIsDragging] = useState(false);
  const [dragOffset, setDragOffset] = useState({ x: 0, y: 0 });
  const switcherRef = useRef<HTMLDivElement>(null);

  const handleMouseDown = (e: React.MouseEvent) => {
    setIsDragging(true);
    setDragOffset({
      x: e.clientX - position.x,
      y: e.clientY - position.y,
    });
  };

  const handleMouseMove = (e: MouseEvent) => {
    if (!isDragging) return;
    
    let newX = e.clientX - dragOffset.x;
    let newY = e.clientY - dragOffset.y;
    
    const maxX = window.innerWidth - (switcherRef.current?.offsetWidth || 200);
    const maxY = window.innerHeight - (switcherRef.current?.offsetHeight || 80);
    
    newX = Math.max(0, Math.min(newX, maxX));
    newY = Math.max(0, Math.min(newY, maxY));
    
    setPosition({ x: newX, y: newY });
  };

  const handleMouseUp = () => {
    setIsDragging(false);
  };

  useEffect(() => {
    if (isDragging) {
      window.addEventListener('mousemove', handleMouseMove);
      window.addEventListener('mouseup', handleMouseUp);
    }
    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mouseup', handleMouseUp);
    };
  }, [isDragging, dragOffset]);

  return (
    <div
      ref={switcherRef}
      className={cn(
        'fixed z-[9999] flex items-center gap-3 px-4 py-3 bg-white/80 backdrop-blur-md rounded-full shadow-lg border border-black/5 transition-shadow',
        isDragging && 'cursor-grabbing shadow-xl'
      )}
      style={{
        left: position.x,
        top: position.y,
        cursor: isDragging ? 'grabbing' : 'grab',
      }}
      onMouseDown={handleMouseDown}
    >
      <span className="text-xs text-gray-500 mr-1">主题</span>
      {themes.map((theme) => (
        <ThemeButton
          key={theme.id}
          theme={theme}
          isActive={themeId === theme.id}
          onClick={() => setTheme(theme.id)}
        />
      ))}
    </div>
  );
}

interface ThemeButtonProps {
  theme: ThemeManifest;
  isActive: boolean;
  onClick: () => void;
}

function ThemeButton({ theme, isActive, onClick }: ThemeButtonProps) {
  return (
    <button
      onClick={(e) => {
        e.stopPropagation();
        onClick();
      }}
      title={theme.name}
      className={cn(
        'relative w-8 h-8 rounded-full transition-all duration-300',
        'hover:scale-110 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-offset-white',
        isActive && 'ring-2 ring-offset-2 ring-offset-white'
      )}
      style={{
        background: isActive 
          ? `linear-gradient(145deg, ${theme.colors.coil.color}40, ${theme.colors.coil.color}20)`
          : theme.colors.bgSurface,
        '--tw-ring-color': theme.colors.accent,
      } as React.CSSProperties}
    >
      {isActive && (
        <div 
          className="absolute inset-0 rounded-full animate-pulse opacity-50"
          style={{ background: `radial-gradient(circle, ${theme.colors.coil.color}30 0%, transparent 70%)` }}
        />
      )}
      <span className="sr-only">{theme.name}</span>
    </button>
  );
}
```

```typescript
/**
 * src/components/shared/FogLayers.tsx
 * 
 * 背景层 + 粒子特效
 * 根据当前主题配置渲染粒子效果
 */

import { useEffect, useRef } from 'react';
import { useThemeStore } from '../../stores/useThemeStore';

export default function FogLayers() {
  const fogRef = useRef<HTMLDivElement>(null);
  const currentTheme = useThemeStore((state) => state.getCurrentTheme());

  useEffect(() => {
    const handleScroll = function() {
      if (fogRef.current) {
        const scrolled = window.pageYOffset;
        fogRef.current.style.transform = 'translateY(' + (scrolled * 0.3) + 'px)';
      }
    };

    window.addEventListener('scroll', handleScroll, { passive: true });
    return function() {
      window.removeEventListener('scroll', handleScroll);
    };
  }, []);

  if (!currentTheme) return null;

  const colors = currentTheme.colors;
  const effects = currentTheme.effects;

  return (
    <div ref={fogRef} className="absolute inset-0 pointer-events-none overflow-hidden">
      {/* 背景渐变 */}
      <div
        className="absolute inset-0"
        style={{
          background: effects?.gradients?.enabled !== false
            ? `
              radial-gradient(ellipse at 20% 30%, ${colors.accent}20 0%, transparent 40%),
              radial-gradient(ellipse at 80% 40%, ${colors.lost.color}20 0%, transparent 40%),
              radial-gradient(ellipse at 50% 80%, ${colors.awaken.color}15 0%, transparent 50%)
            `
            : 'transparent',
          opacity: effects?.gradients?.intensity ?? 0.6,
        }}
      />

      {/* 粒子效果 */}
      {effects?.particles?.enabled && (
        <Particles 
          type={effects.particles.type}
          count={effects.particles.count ?? 20}
          colors={effects.particles.colors ?? [colors.coil.color]}
        />
      )}

      <style>{`
        .particle {
          position: absolute;
          opacity: 0.8;
          animation: particleFall linear infinite;
          pointer-events: none;
        }

        .particle::before {
          content: '';
          position: absolute;
          width: 100%;
          height: 100%;
          background: currentColor;
          border-radius: 50% 0 50% 50%;
          transform: rotate(45deg);
        }

        .particle.sakura {
          box-shadow: 0 0 10px currentColor;
        }

        @keyframes particleFall {
          0% {
            transform: translateY(-10vh) rotate(0deg) translateX(0);
            opacity: 0;
          }
          10% {
            opacity: 0.8;
          }
          90% {
            opacity: 0.6;
          }
          100% {
            transform: translateY(110vh) rotate(720deg) translateX(100px);
            opacity: 0;
          }
        }
      `}</style>
    </div>
  );
}

interface ParticlesProps {
  type: 'sakura' | 'snow' | 'leaves' | 'none';
  count: number;
  colors: string[];
}

function Particles({ type, count, colors }: ParticlesProps) {
  if (type === 'none' || count === 0) return null;

  return (
    <div className="absolute inset-0 overflow-hidden">
      {[...Array(count)].map((_, i) => (
        <div
          key={i}
          className={`particle ${type}`}
          style={{
            left: `${Math.random() * 100}%`,
            color: colors[i % colors.length],
            animationDuration: `${6 + Math.random() * 10}s`,
            animationDelay: `${Math.random() * 10}s`,
            width: `${8 + Math.random() * 12}px`,
            height: `${8 + Math.random() * 12}px`,
            opacity: 0.5 + Math.random() * 0.4,
          }}
        />
      ))}
    </div>
  );
}
```
