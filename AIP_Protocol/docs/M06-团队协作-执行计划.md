## 📋 Week 1: 权限管理系统（9/15 - 9/21）

### 目标
实现基于角色的访问控制（RBAC），确保安全与协作的平衡。

### 为什么需要权限管理？

**问题**：
- 所有人都能改所有东西 → 容易误操作
- 新成员不知道能做什么
- 无法限制生产环境访问
- 审计时不知道谁动了什么

**解决方案**：
```
角色定义：
┌─────────────┬─────────────────────────────────────┐
│ 角色         │ 权限                                │
├─────────────┼─────────────────────────────────────┤
│ Admin       │ 一切权限（审批决策、删项目）         │
│ Maintainer  │ 创建/分配任务、合并 PR、发布         │
│ Contributor │ 完成任务、创建 PR、评论              │
│ Viewer      │ 只读权限（查看所有，不能修改）       │
└─────────────┴─────────────────────────────────────┘
```

**权限检查**：
```javascript
// auth/permission-checker.js
class PermissionChecker {
  constructor(user) {
    this.user = user;  // { id, name, role, projects }
    this.roles = loadRoles();
  }

  can(permission, resource = null) {
    const rolePermissions = this.roles[this.user.role].permissions;

    // 通配符支持
    const pattern = permission.replace(':*', '.*');

    return rolePermissions.some(p => p.match(pattern));
  }

  // 检查资源级别权限（如：只能访问自己项目的任务）
  canAccess(resourceType, resourceId) {
    if (this.can(`*:*`)) return true;  // admin 拥有所有权限

    if (resourceType === 'task') {
      const task = getTask(resourceId);
      return this.user.projects.includes(task.project_id);
    }

    if (resourceType === 'project') {
      return this.user.projects.includes(resourceId);
    }

    return false;
  }
}

// 使用
const checker = new PermissionChecker(currentUser);

if (!checker.can('task:complete', taskId)) {
  throw new Error('无权完成此任务');
}
```

**认证流程**：
```
登录页面
    ↓ 提交 username/password
验证凭据
    ↓
生成 JWT Token
    ↓
存储到 localStorage
    ↓
后续请求携带 Token（Authorization: Bearer xxx）
    ↓
中间件验证 Token 并设置 req.user
```

**实现**：
```javascript
// auth/index.js
import jwt from 'jsonwebtoken';

const JWT_SECRET = process.env.JWT_SECRET || 'change-me';

// 登录
async function login(username, password) {
  const user = await verifyCredentials(username, password);

  if (!user) {
    throw new Error('用户名或密码错误');
  }

  const token = jwt.sign(
    {
      userId: user.id,
      role: user.role,
      projects: user.projects
    },
    JWT_SECRET,
    { expiresIn: '7d' }
  );

  return { token, user };
}

// 中间件
function authMiddleware(req, res, next) {
  const authHeader = req.headers.authorization;

  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ error: '未授权' });
  }

  const token = authHeader.slice(7);

  try {
    const decoded = jwt.verify(token, JWT_SECRET);
    req.user = decoded;
    next();
  } catch (err) {
    return res.status(401).json({ error: 'Token 无效' });
  }
}
```

```javascript
  const pr = payload.pull_request;
```

```javascript
  const questionEmbedding = await generateEmbedding(question);

  // 2. 向量搜索（k=5）
  const results = await esClient.search({
    index: 'aip-knowledge',

      size: 5,
      query: {
        script_score: {
          query: { match_all: {} },
          script: {
            source: "cosineSimilarity(params.query_vector, 'embedding') + 1.0",
            params: { query_vector: questionEmbedding }

        }
      }
    }

${contexts.map((c, i) => `[${i+1}] ${c}`).join('\n\n')}

请基于上述历史记录回答，不要编造。
如果历史记录不相关，请说"我没有找到相关信息"。
`;

  // 4. 调用 LLM（或使用本地模型）
  const answer = await callLLM(prompt);

  return {

    answer,
    sources: results.hits.hits.map(hit => ({
```
