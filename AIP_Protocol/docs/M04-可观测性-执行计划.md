```javascript
// metrics/sdk.js
class MetricsCollector {
  constructor() {
    this.metrics = new Map();
    this.histograms = new Map();
  }

  // 计数器
  counter(name, labels = {}, value = 1) {
    const key = this.key(name, labels);
    if (!this.metrics.has(key)) {
      this.metrics.set(key, 0);
    }
    this.metrics.set(key, this.metrics.get(key) + value);
  }

  // 仪表盘（瞬时值）
  gauge(name, value, labels = {}) {
    this.metrics.set(this.key(name, labels), value);
  }

  // 直方图（分布）
  histogram(name, value, buckets, labels = {}) {
    // 将 value 归入 bucket
    const bucketKey = this.key(`${name}_bucket`, { ...labels, bucket: this.findBucket(value, buckets) });
    this.counter(bucketKey, labels);

    // 记录总和与计数
    this.counter(`${name}_sum`, labels, value);
    this.counter(`${name}_count`, labels);
  }

  // 导出为 Prometheus 格式
  exportPrometheus() {
    let output = [];
    for (const [key, value] of this.metrics) {
      output.push(`# TYPE ${key.split('{')[0]} counter\n${key} ${value}`);
    }
    return output.join('\n');
  }

  key(name, labels) {
    const labelStr = Object.entries(labels)
      .map(([k, v]) => `${k}="${v}"`).join(',');
    return labelStr ? `${name}{${labelStr}}` : name;
  }

  findBucket(value, buckets) {
    for (const bucket of buckets) {
      if (value <= bucket) return bucket;
    }
    return '+Inf';
  }
}

// 全局单例
export const metrics = new MetricsCollector();

// 使用示例
metrics.gauge('active_projects', 3);
metrics.counter('task_completed', { project: 'moodify' });
metrics.histogram('task_execution_seconds', 120, [60, 300, 600]);
```

**告警引擎**：
```javascript
// monitoring/alert-engine.js
class AlertEngine {
  constructor(rules, notifier) {
    this.rules = rules;
    this.notifier = notifier;
    this.activeAlerts = new Map();
  }

  async evaluate(metrics) {
    for (const rule of this.rules) {
      const result = await this.evaluateRule(rule, metrics);

      if (result.alerting) {
        await this.handleAlert(rule, result);
      } else {
        await this.handleResolve(rule);
      }
    }
  }

  async handleAlert(rule, result) {
    const alertId = `${rule.name}-${result.labels}`;

    if (!this.activeAlerts.has(alertId)) {
      // 新告警
      this.activeAlerts.set(alertId, {
        rule: rule.name,
        startedAt: Date.now(),
        labels: result.labels
      });

      // 发送通知
      await this.notifier.send({
        level: rule.severity,
        title: rule.summary,
        message: rule.description,
        labels: result.labels
      });
    }
  }

  async handleResolve(rule) {
    // 查找并清除告警
    for (const [id, alert] of this.activeAlerts) {
      if (alert.rule === rule.name) {
        await this.notifier.send({
          level: 'info',
          title: `Resolved: ${rule.summary}`,
          message: `告警已解决`
        });
        this.activeAlerts.delete(id);
      }
    }
  }
}
```

```javascript
// 原来：每次 metric 都写文件
// 优化：每 10 秒批量写入
class BufferedMetricsCollector {
  constructor(flushInterval = 10000) {
    this.buffer = [];
    this.flushInterval = flushInterval;
    setInterval(() => this.flush(), flushInterval);
  }

  record(metric) {
    this.buffer.push(metric);
    // 超过 1000 条立即刷新
    if (this.buffer.length > 1000) this.flush();
  }

  flush() {
    if (this.buffer.length === 0) return;

    // 批量写入
    fs.appendFileSync(
      'metrics/current.jsonl',
      this.buffer.map(m => JSON.stringify(m)).join('\n') + '\n'
    );

    this.buffer = [];
  }
}
```
