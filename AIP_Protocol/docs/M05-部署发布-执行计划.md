```javascript
// deploy/config-loader.js
class ConfigLoader {
  constructor(env = 'development') {
    this.env = env;
    this.config = this.load(env);
  }

  load(env) {
    const environments = readYAML('deploy/environments.yaml');
    const config = environments[env];

    if (!config) {
      throw new Error(`未知环境: ${env}`);
    }

    // 合并基础配置
    return deepMerge(environments.base, config);
  }

  get(key) {
    return getNestedValue(this.config, key);
  }

  // 环境变量覆盖
  applyOverrides() {
    for (const [key, value] of Object.entries(process.env)) {
      if (key.startsWith('MOODIFY_')) {
        const configKey = key.slice(8).toLowerCase().replace(/_/g, '.');
        this.set(configKey, value);
      }
    }
  }
}
```

```javascript
const schema = {
  type: 'object',
  properties: {
    api_url: { type: 'string', format: 'uri' },
    database: {
      type: 'object',
      required: ['type', 'host', 'port']
    },
    logging: {
      type: 'object',
      properties: {
        level: { enum: ['debug', 'info', 'warn', 'error'] }
      }
    }
  },
  required: ['api_url', 'database']
};

function validateConfig(config) {
  const errors = [];

  // 使用 AJV 验证
  const valid = ajv.validate(schema, config);
  if (!valid) {
    errors.push(...ajv.errors);
  }

  // 自定义规则
  if (config.database.port < 1 || config.database.port > 65535) {
    errors.push('数据库端口必须在 1-65535 之间');
  }

  return { valid: errors.length === 0, errors };
}
```

```javascript
// tools/generate-release-checklist.js
function generateChecklist(commitRange) {
  const changedFiles = getChangedFiles(commitRange);

  // 分析改动类型
  const changes = analyzeChanges(changedFiles);

  // 加载模板
  const template = loadYAML('templates/release-checklist.yaml');

  // 筛选需要的项
  const checklist = {
    version: getLatestTag(),
    date: new Date().toISOString(),
    sections: []
  };

  for (const section of template.sections) {
    const items = section.items.filter(item => {
      // 检查条件
      if (item.condition === 'always') return true;
      if (item.condition === 'has_db_migration' && changes.hasDbMigration) return true;
      if (item.condition === 'changed(frontend/)' && changes.changed('frontend/')) return true;
      return false;
    });

    if (items.length > 0) {
      checklist.sections.push({ ...section, items });
    }
  }

  return checklist;
}

// 生成 Markdown
function renderMarkdown(checklist) {
  let md = `# 发布清单 - ${checklist.version} (${checklist.date})\n\n`;

  for (const section of checklist.sections) {
    md += `## ${section.name}\n\n`;
    for (const item of section.items) {
      const status = item.required ? '[ ]' : '[ ] (可选)';
      md += `${status} ${item.description}\n`;
    }
    md += '\n';
  }

  return md;
}
```

```javascript
// tools/release-manager.js
async function interactiveRelease() {
  const checklist = generateChecklist();

  for (const section of checklist.sections) {
    console.log(`\n${section.name}`);

    for (const item of section.items) {
      const prompt = `  [ ] ${item.description} (y/n/s) `;
      const answer = await askQuestion(prompt);

      if (answer === 'y') {
        console.log(`  > 执行: ${item.command}`);
        const result = await exec(item.command);

        if (result.success) {
          item.status = 'completed';
          item.completedAt = new Date();
          console.log(`  ✅ 完成`);
        } else {
          item.status = 'failed';
          console.log(`  ❌ 失败: ${result.error}`);
          // 选择：继续、跳过、中止
          const action = await askQuestion('  继续? (continue/skip/abort): ');
          if (action === 'abort') {
            console.log('❌ 发布中止');
            process.exit(1);
          }
        }
      }
    }
  }

  // 生成报告
  saveReport(checklist);
}
```

## 📋 Week 3: 回滚与状态追踪（8/29 - 9/4）

### 目标
当发布失败时，能快速回滚到上一个稳定版本。

### 回滚策略

#### 3 层回滚机制

```
发布失败
    ↓
Layer 1: 应用级回滚（快速）
  → 切换流量到旧版本
  → 零停机（如果蓝绿部署）

    ↓ 仍失败
Layer 2: 数据级回滚
  → 回滚数据库迁移
  → 恢复数据快照

    ↓ 仍失败
Layer 3: 基础设施回滚
  → 恢复整个环境快照
  → 从备份还原
```

```javascript
// deploy/rollback.js
async function rollback(version) {
  console.log(`🔄 开始回滚到 ${version}...`);

  // 1. 检查当前版本
  const current = await getCurrentVersion();
  console.log(`   当前: ${current} → 目标: ${version}`);

  // 2. 切换流量（Kubernetes）
  await exec(`kubectl set image deployment/backend backend=moodify/backend:${version}`);
  await exec(`kubectl set image deployment/frontend frontend=moodify/frontend:${version}`);

  // 3. 等待新版本就绪
  await waitForReady('backend', 300);
  await waitForReady('frontend', 300);

  // 4. 健康检查
  const healthy = await checkHealth();
  if (!healthy) {
    throw new Error('回滚后健康检查失败');
  }

  // 5. 记录回滚事件
  logEvent({
    type: 'rollback',
    from: current,
    to: version,
    timestamp: new Date()
  });

  console.log(`✅ 回滚完成`);
}

// 一键回滚到上一个版本
async function rollbackToLast() {
  const lastRelease = await getLastSuccessfulRelease();
  await rollback(lastRelease.version);
}
```

```javascript
// deploy/status-tracker.js
class DeploymentTracker {
  constructor() {
    this.history = loadHistory();
  }

  async recordDeployment(deployment) {
    this.history.unshift({
      id: generateId(),
      ...deployment,
      timestamp: new Date().toISOString()
    });

    saveHistory(this.history);

    // 更新仪表盘
    eventEmitter.emit('deployment:completed', deployment);
  }

  getLastSuccessful(environment = 'production', count = 5) {
    return this.history
      .filter(d => d.environment === environment && d.status === 'success')
      .slice(0, count);
  }

  getSuccessRate(environment, days = 30) {
    const since = Date.now() - days * 24 * 60 * 60 * 1000;
    const recent = this.history.filter(d =>
      d.environment === environment &&
      new Date(d.timestamp) > since
    );

    if (recent.length === 0) return 0;

    const success = recent.filter(d => d.status === 'success').length;
    return (success / recent.length) * 100;
  }
}
```

```javascript
// deploy/policy-enforcer.js
async function checkReleasePolicy(release) {
  const policy = loadPolicy();

  // 检查是否在发布窗口
  if (!isInWindow(policy.windows)) {
    throw new Error('当前不在发布窗口内');
  }

  // 检查是否需要审批
  if (policy.require_approval && !hasApproval(release)) {
    // 检查自动审批规则
    if (await evaluateAutoApprovalRules(policy.auto_approval_rules, release)) {
      console.log('✅ 满足自动审批条件，跳过人工审批');
    } else {
      throw new Error('需要人工审批，请@tech-lead 和 @pm');
    }
  }
}
```
